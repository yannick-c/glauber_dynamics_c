116 graph *graph_construct_torus(int n, int d, int init_weight){
117  g_assert(n>2); /* if n=2 then you get connections like - 0 - 1 - which
118  is a double edge so no periodic boundary conditions are possible
119  */
120  graph *out = graph_new();
121  int vertex_count = int_pow(n, d);
122  graph_add_n_vertices(out, vertex_count);
123 
124  for (int i=0; i<vertex_count; i++){
125  for (int j=0; j<d; j++){
126  /* the offset ensures that we dont modulo down to the
127  * first cases all the time but only close the
128  * boundaries */
129  int offset = i - (i % int_pow(n,j+1));
130  /* connect it to the next vertex in the next
131  * dimensions, i.e. in 2 dimensions the lower one and
132  * the right one take care of the periodic boundary conditions. */
133  int connect_to = (i + (int_pow(n,j))) % int_pow(n,j+1) + offset;
134  graph_add_edge(out, i, connect_to, init_weight);
135  }
136  }
137  return out;
138 }
