19 /* get an exponential random variable */
20 static double exponential_rand(double lambda_rate){
21  /* first generate a 'double' using the recipe in the docs for the C
22  * implementation */
23  double unif_dbl = ldexp(pcg32_random_r(&exponential_rng), -32);
24  /* for the following line consider that -log(U)/lambda_rate with U
25  * uniform on [0, 1) gives an exponential distribution */
26  return - log(unif_dbl)/lambda_rate;
27 }

29 /*
30  * For the implementation of poisson clocks on every vertex use that the
31  * time between events of a Poisson point process of rate 1 is the exponential
32  * distribution with rate 1. Furthermore, for n vertices with independent
33  * exponential distribution clocks {exp_i}_{i\in [0,...,n-1]} on them the
34  * next time something fires has distribution min({exp_i}) which is again
35  * exponentially distributed with rate n (i.e. the sum of rates).
36  *
37  * Hence, we can take a single exponential time and then choose the vector
38  * uniformly which is less computatinally intensive than putting an exponential
39  * clock on every vertex and managing their order.
40  *
41  * NOTE: The RNG is initialized only here so glauber_dynamics has to be called
42  * as the first and only function of this file.
43  */
44 void glauber_dynamics(graph *init_state,
45  update_rule graph_update,
46  int threshold_time,
47  arguments *args){
48  double t = 0; /* time parameter */
49 
50  /* start RNG initialization */
51  uint64_t seeds1[2], seeds2[2], seeds3[2];
52  entropy_getbytes((void*)seeds1, sizeof(seeds1));
53  entropy_getbytes((void*)seeds2, sizeof(seeds2));
54  entropy_getbytes((void*)seeds3, sizeof(seeds3));
55  pcg32_srandom_r(&exponential_rng, seeds1[0], seeds1[1]);
56  pcg32_srandom_r(&uniform_rng, seeds2[0], seeds2[1]);
57  pcg32_srandom_r(&update_rng, seeds3[0], seeds3[1]);
58  /* end RNG initialization */
59 
60  double prev_frame = 0; // when the previous frame was drawn
61  while (t < threshold_time){
62  double cur_time = exponential_rand((double) init_state->n);
63  t += cur_time;
64 
65  /* it generates strictly smaller than bound so init_state->n is
66  * fine. */
67  int chosen_vertex_index = pcg32_boundedrand_r(&uniform_rng,
68  init_state->n);
69 
70  /* use the passed graph_update rule to update the graph state */
71  graph_update(init_state, chosen_vertex_index, args->alpha,
72  &update_rng);
73 
74  if (!args->silent && t-prev_frame>args->frame_density){
75  draw_torus2png(init_state, args->n, args->d, round((t-prev_frame)),
76  NULL, args->width, args->height, args->dpi,
77  args->penwidth, args->decrease_rate);
78  prev_frame=t;
79  }
80  }
